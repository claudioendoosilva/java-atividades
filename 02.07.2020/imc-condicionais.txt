public class Main
{

public static void main(String[] args) 
{

//var

 double altura,peso,imc;
 
//entrada

altura= Double.parseDouble(JOptionPane.showInputDialog("Digite sua altura"));

peso= Double.parseDouble(JOptionPane.showInputDialog("Digite seu peso"));

//processamento

imc=(peso/(altura*altura));
 
if (imc<=18.5)
{

JOptionPane.showMessageDialog(null, "Está abaixo do peso");
 
       }

 if ((imc>=18.6) && (imc<=24.9))
{
  JOptionPane.showMessageDialog(null, "Peso normal");
  
 }
  
  if ((imc>=25) && (imc<=29.9))
{
    
 JOptionPane.showMessageDialog(null, "Está acima do peso");
  
      }
     
if ((imc>=30) && (imc<=34.9))
{
       
 JOptionPane.showMessageDialog(null, "Está na obesidade de grau 1");

        }
        
 
  if ((imc>=35) && (imc<=39.9))
{
          
  JOptionPane.showMessageDialog(null, "Está na obesidade de grau 2");
 
       }
   
  if (imc>=40)
{
           
 JOptionPane.showMessageDialog(null, "Está na obesidade de grau 3");

        }
        
   
JOptionPane.showMessageDialog(null, "Seu IMC é"+imc);
	
}

}
